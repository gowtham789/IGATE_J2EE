27. CREATE VIEW ACC_VIEW(CUSTOMERCODE, ACCOUNTHOLDERNAME, ACCOUNTNUMBER, TYPE, BALANCE)
	AS
	SELECT A.CUSTOMERID, C.CUSTOMERNAME, A.ACCOUNTNUMBER, A.ACCOUNTTYPE, A.LEDGERBALANCE
	FROM ACCOUNTMASTER A, CUSTOMERMASTER C
	WHERE A.CUSTOMERID=C.CUSTOMERID;
	
28. CREATE VIEW VACCS_DTLS(ID,NAME,TYPE,BALANCE)
	AS

	FROM ACCOUNTMASTER A, CUSTOMERMASTER C
	WHERE A.CUSTOMERID=C.CUSTOMERID
