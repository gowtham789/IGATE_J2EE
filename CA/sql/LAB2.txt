
2.1   SELECT STAFF_NAME, LPAD(STAFF_SAL,15,'$')
		FROM STAFF_MASTER
		/

2.2   SELECT STUDENT_NAME, TO_CHAR(STUDENT_DOB, 'MONTH,DD YYYY')
	FROM STUDENT_MASTER
	WHERE (to_char(STUDENT_DOB, 'DY') IN ('SAT','SUN'));
	/
	
2.3 SELECT STAFF_NAME, ROUND(MONTHS_BETWEEN(SYSDATE,HIREDATE)) AS MONTHS_WORKED
	FROM STAFF_MASTER
	ORDER BY MONTHS_WORKED
	/
	
2.4 SELECT STAFF_NAME, STAFF_SAL, RPAD(' ', (STAFF_SAL/1000)+1,'X') XSAL
	FROM STAFF_MASTER;
	
2.5 SELECT STAFF_NAME, HIREDATE
	FROM STAFF_MASTER
	WHERE EXTRACT(MONTH FROM HIREDATE) IN (12)
	AND EXTRACT(DAY FROM HIREDATE)<15	

2.6 SELECT STAFF_NAME, STAFF_SAL,
    CASE 
	WHEN STAFF_SAL>=50000 THEN 'A    '
	WHEN STAFF_SAL>=25000 AND STAFF_SAL<50000 THEN 'B    '
    WHEN STAFF_SAL>=10000 AND STAFF_SAL<25000 THEN 'C    '
	ELSE 'D    '
	END GRADE
	FROM STAFF_MASTER;
     /

2.7 SELECT STAFF_NAME, HIREDATE,
	TO_CHAR(HIREDATE, 'DAY') "DAY OF THE WEEK"
	FROM STAFF_MASTER
	ORDER BY MOD(TO_CHAR(HIREDATE, 'D')+5,7)
	/

2.8 SELECT STAFF_NAME NAME,
	RPAD(
	  (RPAD((SUBSTR(STAFF_NAME,1,1)),(LENGTH(STAFF_NAME)-1),'*')),
	  LENGTH(STAFF_NAME),
	  SUBSTR(STAFF_NAME, LENGTH(STAFF_NAME),LENGTH(STAFF_NAME))
	  )
	  ASTERIX
	  FROM STAFF_MASTER;
	  
2.9  SELECT INSTR('MISSISSIPI','I',2,3) FROM DUAL;

2.10 SELECT TO_CHAR(NEXT_DAY((LAST_DAY(SYSDATE)),'FRIDAY')-7,'DD MONTH,YYYY')
		FROM DUAL;

2.12  SELECT STUDENT_NAME, DEPT_CODE,
